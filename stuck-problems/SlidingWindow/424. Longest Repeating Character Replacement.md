## 5) Code (final)

```python
class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        res=0
        l=0
        count={}
        max_count=0
        for r in range(len(s)):
            if s[r] not in count:
                count[s[r]]=1
            else:
                count[s[r]]+=1
            max_count=max(max_count, count[s[r]])
            while (r-l+1) - max_count>k:
                count[s[l]]-=1
                l+=1
            res=max(res, r-l+1)
        return res
```
