## 5) Code (final)

```python
class Codec:
    def encode(self, strs: List[str]) -> str:
        """Encodes a list of strings to a single string.
        """
        res=""
        for word in strs:
            res+=str(len(word))+"#"+word
        return res

    def decode(self, s: str) -> List[str]:
        """Decodes a single string to a list of strings.
        """
        i=0
        res=[] #array to get the output out
        while i<len(s):
            j=i
            while s[j]!="#":
                j=j+1
            length=int(s[i:j])
            i=j+1
            j=length+i
            res.append(s[i:j])
            i=j
        return res
```
