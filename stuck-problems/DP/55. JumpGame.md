## 5) Code (final)

class Solution:
    def canJump(self, nums: List[int]) -> bool:
        max_reach=0
        last=len(nums)-1

        for i, jump in enumerate(nums):
            #smaller max reach at the index
            if i>max_reach:
                return False
            max_reach=max(max_reach, i+jump)
            #at index max_reach number is greater or equal to last index
            if max_reach>=last:
                return True
        return True
