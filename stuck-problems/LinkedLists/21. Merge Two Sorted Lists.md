## 5) Code (final)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        # we need seperate list so use dummy
        #updating the pointers?
        dummy=ListNode(-1)
        curr=dummy
        if list1 and not list2:
            return list1
        if list2 and not list1:
            return list2
        while list1 and list2:
            if list1.val<=list2.val:
                curr.next=list1
                list1=list1.next
            else:
                curr.next=list2
                list2=list2.next
            curr=curr.next
        if list1:
            curr.next=list1
            return dummy.next
        if list2:
            curr.next=list2
            return dummy.next
