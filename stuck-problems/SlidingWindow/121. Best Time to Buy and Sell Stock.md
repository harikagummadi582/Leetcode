## 5) Code (final)

class Solution:
def maxProfit(self, prices: List[int]) -> int:
max_profit=0
finalMinPrice=prices[0]
for i in range(len(prices)):
finalMinPrice=min(prices[i], finalMinPrice)
curr_profit=prices[i]-finalMinPrice
max_profit=max(max_profit,curr_profit)

        return max_profit
