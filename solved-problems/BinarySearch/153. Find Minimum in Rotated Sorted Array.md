```python
class Solution:
    def findMin(self, nums: List[int]) -> int:
        l=0
        r=len(nums)-1
        min_num=float('inf')
        while l<=r:
            mid=(l+r)//2
            min_num=min(min_num, nums[mid], nums[l], nums[r])
            print(min_num)
            if nums[l]<=min_num<nums[mid]:
                r=mid
            else:
                l=l+1
        return min_num
```
